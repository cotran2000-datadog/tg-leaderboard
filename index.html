const API='https://script.google.com/macros/s/AKfycbyY0_NEbZF6EXccgI5ri4StsSfbPL7fWURbKF4HphzLKvCjxWhBXoD2wXrlqTtj2Q0/exec';

const ADMIN_PIN='1234';

let entries=[];

function parseTime(v){

  v=v.trim();

  let m=v.match(/^(\d{1,2}):(\d{2})(?:\.(\d{1,3}))?$/);

  if(m)return+m[1]*60000+ +m[2]*1000+parseInt((m[3]||'0').padEnd(3,'0'));

  m=v.match(/^(\d+)(?:\.(\d{1,3}))?$/);

  if(m)return+m[1]*1000+parseInt((m[2]||'0').padEnd(3,'0'));

  return null;

}

function fmt(ms){

  const m=Math.floor(ms/60000),s=Math.floor(ms%60000/1000),cs=Math.floor(ms%1000/10);

  return m>0?`${m}:${String(s).padStart(2,'0')}.${String(cs).padStart(2,'0')}`:`${s}.${String(cs).padStart(2,'0')}`;

}

function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')}

function sorted(){return [...entries].sort((a,b)=>a.ms-b.ms)}


async function fetchEntries(){

  const res=await fetch(API+'?t='+Date.now());

  const data=await res.json();

  entries=data.map(e=>({...e,ms:Number(e.ms)}));

}


async function postEntry(entry){

  await fetch(API,{

    method:'POST',

    mode:'no-cors',

    headers:{'Content-Type':'application/json'},

    body:JSON.stringify(entry)

  });

}


async function pollEntries(){

  try{

    await fetchEntries();

    if(currentView==='dashboard')renderApp();

  }catch(e){console.warn('Poll failed',e)}

}


async function doSubmit(){

  const n=(document.getElementById('inp-name')?.value||'').trim();

  const t=(document.getElementById('inp-time')?.value||'').trim();

  const err=document.getElementById('err');

  if(!n){err.style.display='block';err.textContent='Enter your name';return}

  const ms=parseTime(t);

  if(!ms||ms<=0){err.style.display='block';err.textContent='Enter a valid time — e.g. 1:23.45 or 83.45';return}

  const btn=document.getElementById('submit-btn');

  if(btn){btn.textContent='Saving...';btn.disabled=true;}

  try{

    const entry={id:Date.now(),name:n,ms,ts:Date.now()};

    // Optimistic local update

    const idx=entries.findIndex(e=>e.name.toLowerCase()===n.toLowerCase());

    if(idx>=0)entries[idx]=entry;else entries.push(entry);

    await postEntry(entry);

    // no-cors means we can't read response — wait briefly then re-fetch

    await new Promise(r=>setTimeout(r,1200));

    await fetchEntries();

    submittedName=n;

    renderApp();

  }catch(e){

    if(btn){btn.textContent='Set Time';btn.disabled=false;}

    const err2=document.getElementById('err');

    if(err2){err2.style.display='block';err2.textContent='Sync failed — check connection';}

  }

}


async function doReset(){

  const p=document.getElementById('pin-inp')?.value;

  if(p===ADMIN_PIN){

    // Post a special reset command

    await fetch(API,{method:'POST',mode:'no-cors',headers:{'Content-Type':'application/json'},body:JSON.stringify({reset:true})});

    await new Promise(r=>setTimeout(r,1200));

    await fetchEntries();

    hideAdminModal();

    renderApp();

  }else{

    const e=document.getElementById('pin-err');

    if(e)e.style.display='block';

  }

}


function showAdminModal(){

  const m=document.getElementById('admin-modal');

  if(m){m.style.display='flex';document.getElementById('pin-inp').value='';document.getElementById('pin-err').style.display='none';}

}

function hideAdminModal(){

  const m=document.getElementById('admin-modal');

  if(m)m.style.display='none';

}

    if(qrEl) new QRCode(qrEl,{text:location.href.split('?')[0],width:100,height:100,colorDark:'#ffffff',colorLight:'#1a1a1a'});

  }

}
